--
-- CHAR
--

-- Per SQL standard, CHAR means character(1), that is a varlena type
-- with a constraint restricting it to one character (not byte)

SELECT char 'c' = char 'c' AS true;
RESULT:
	postgres: [(True,)]

CREATE TEMP TABLE CHAR_TBL(f1 char);
RESULT:
	postgres: None

INSERT INTO CHAR_TBL (f1) VALUES ('c     ');

QUERY:



SELECT * FROM CHAR_TBL;
RESULT:
	postgres: [('a',), ('A',), ('1',), ('2',), ('3',), (' ',), ('c',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 <> 'a';
RESULT:
	postgres: [('A',), ('1',), ('2',), ('3',), (' ',), ('c',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 = 'a';
RESULT:
	postgres: [('a',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 < 'a';
RESULT:
	postgres: [('A',), ('1',), ('2',), ('3',), (' ',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 <= 'a';
RESULT:
	postgres: [('a',), ('A',), ('1',), ('2',), ('3',), (' ',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 > 'a';
RESULT:
	postgres: [('c',)]

-----------
QUERY:


SELECT c.*
   FROM CHAR_TBL c
   WHERE c.f1 >= 'a';
RESULT:
	postgres: [('a',), ('c',)]

--
-- Also test /* REPLACED */''char/* REPLACED */'', which is an ad-hoc one-byte type.  It can only
-- really store ASCII characters, but we allow high-bit-set characters
-- to be accessed via bytea-like escapes.
--

SELECT 'a'::"char";
RESULT:
	postgres: [('a',)]

-----------
QUERY:

SELECT '\101'::"char";
RESULT:
	postgres: [('A',)]

-----------
QUERY:

SELECT '\377'::"char";
RESULT:
	postgres: [('\\377',)]

-----------
QUERY:

SELECT 'a'::"char"::text;

QUERY:
